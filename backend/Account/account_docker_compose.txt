version: '3.8'

services:
  account-service:
    build: .
    ports:
      - "3002:3002"
    environment:
      - NODE_ENV=production
      - PORT=3002
      - MONGODB_URI=mongodb://account-mongo:27017/banking_accounts
      - AUTH_SERVICE_URL=http://auth-service:3001
      - ALLOWED_ORIGINS=http://localhost:3000
      - DEFAULT_CURRENCY=USD
    depends_on:
      - account-mongo
    networks:
      - account-network
      - banking-network
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "node", "src/healthcheck.js"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s

  account-mongo:
    image: mongo:7.0
    ports:
      - "27018:27017"
    environment:
      - MONGO_INITDB_DATABASE=banking_accounts
    volumes:
      - account-mongo-data:/data/db
      - ./mongo-init.js:/docker-entrypoint-initdb.d/mongo-init.js:ro
    networks:
      - account-network
    restart: unless-stopped
    healthcheck:
      test: echo 'db.runCommand("ping").ok' | mongosh localhost:27017/banking_accounts --quiet
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s

  account-mongo-express:
    image: mongo-express:1.0.0
    ports:
      - "8082:8081"
    environment:
      - ME_CONFIG_MONGODB_SERVER=account-mongo
      - ME_CONFIG_MONGODB_PORT=27017
      - ME_CONFIG_MONGODB_ENABLE_ADMIN=true
      - ME_CONFIG_BASICAUTH_USERNAME=admin
      - ME_CONFIG_BASICAUTH_PASSWORD=admin123
    depends_on:
      - account-mongo
    networks:
      - account-network
    restart: unless-stopped
    profiles:
      - dev

networks:
  account-network:
    driver: bridge
  banking-network:
    external: true

volumes:
  account-mongo-data:
    driver: local
